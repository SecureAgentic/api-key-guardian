# === API Key Guardian — Cloud Run Service ===
# File: guardian_proxy/main.py

import os, time, json, logging
from flask import Flask, request, jsonify, abort
from collections import defaultdict
from google.cloud import apikeys_v2, pubsub_v1
import httpx

app = Flask(__name__)
logging.basicConfig(level=logging.INFO)

# === Config (set via env vars in Cloud Run) ===
RATE_LIMIT_RPM       = int(os.environ.get("RATE_LIMIT_RPM", "100"))
WARN_BUDGET_USD      = float(os.environ.get("WARN_BUDGET_USD", "10"))
BLOCK_BUDGET_USD     = float(os.environ.get("BLOCK_BUDGET_USD", "100"))
PROJECT_ID           = os.environ.get("PROJECT_ID", "")
ALERT_TOPIC          = os.environ.get("ALERT_TOPIC", "api-guardian-alerts")
API_KEY_ID           = os.environ.get("MANAGED_KEY_ID", "")

# === In-memory tracking (use Redis in production) ===
ip_requests   = defaultdict(list)   # IP → [timestamps]
blocked_ips   = set()              # Permanently blocked IPs
daily_spend   = 0.0               # Track today's USD spend
total_blocked = 0

# ─────────────────────────────────────────────
def is_rate_limited(ip: str) -> bool:
    now = time.time()
    window = ip_requests[ip]
    # Keep only requests within the last 60 seconds
    ip_requests[ip] = [t for t in window if now - t < 60]
    if len(ip_requests[ip]) >= RATE_LIMIT_RPM:
        return True
    ip_requests[ip].append(now)
    return False

def publish_alert(alert_type: str, details: dict):
    try:
        publisher = pubsub_v1.PublisherClient()
        topic = f"projects/{PROJECT_ID}/topics/{ALERT_TOPIC}"
        data = json.dumps({"type": alert_type, "ts": time.time(), **details})
        publisher.publish(topic, data.encode())
        logging.info(f"Alert published: {alert_type}")
    except Exception as e:
        logging.error(f"Alert failed: {e}")

def revoke_api_key():
    try:
        client = apikeys_v2.ApiKeysClient()
        key_name = f"projects/{PROJECT_ID}/locations/global/keys/{API_KEY_ID}"
        key = client.get_key(name=key_name)
        key.restrictions.api_targets = []
        client.update_key(key=key, update_mask={"paths": ["restrictions"]})
        logging.critical("🔒 API KEY REVOKED — budget exceeded")
    except Exception as e:
        logging.error(f"Key revocation failed: {e}")

# ─────────────────────────────────────────────
@app.route("/v1/proxy", methods=["POST"])
def proxy_request():
    global daily_spend, total_blocked
    client_ip = request.headers.get("X-Forwarded-For", request.remote_addr)

    # 1. Check blocked IP list
    if client_ip in blocked_ips:
        total_blocked += 1
        abort(429, "IP permanently blocked by Guardian")

    # 2. Enforce rate limit
    if is_rate_limited(client_ip):
        blocked_ips.add(client_ip)
        total_blocked += 1
        publish_alert("RATE_LIMIT_BREACH", {"ip": client_ip, "rpm": RATE_LIMIT_RPM})
        abort(429, "Rate limit exceeded — IP auto-blocked")

    # 3. Check budget cap
    if daily_spend >= BLOCK_BUDGET_USD:
        publish_alert("BUDGET_EXCEEDED", {"spend": daily_spend, "limit": BLOCK_BUDGET_USD})
        revoke_api_key()
        abort(402, "Daily budget exceeded — key auto-revoked")

    # 4. Forward to Gemini API
    api_key = os.environ.get("GEMINI_API_KEY")
    resp = httpx.post(
        "https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent",
        params={"key": api_key},
        json=request.get_json(),
        timeout=30
    )

    # 5. Track cost (estimate based on token count)
    result = resp.json()
    tokens = result.get("usageMetadata", {}).get("totalTokenCount", 0)
    daily_spend += tokens * 0.000001  # $1 per 1M tokens
    if daily_spend >= WARN_BUDGET_USD:
        publish_alert("BUDGET_WARNING", {"spend": daily_spend, "limit": WARN_BUDGET_USD})

    return jsonify(result), resp.status_code

@app.route("/health")
def health():
    return jsonify({"status": "ok", "blocked_ips": len(blocked_ips), "spend": daily_spend})

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=int(os.environ.get("PORT", 8080)))

@functions_framework.cloud_event
def budget_kill_switch(cloud_event):
    # Decode Pub/Sub budget notification
    data = json.loads(base64.b64decode(cloud_event.data["message"]["data"]))
    cost_amount  = data.get("costAmount", 0)
    budget_amount = data.get("budgetAmount", 999)
    alert_threshold = data.get("alertThresholdExceeded", 0)

    print(f"💰 Budget alert: ${cost_amount:.2f} of ${budget_amount:.2f} ({alert_threshold*100:.0f}%)")

    # Warning level: $10 hit → log + alert only
    if cost_amount >= WARN_BUDGET and cost_amount < BLOCK_BUDGET:
        print("⚠️  WARNING threshold hit — sending alerts")
        _create_monitoring_incident("WARNING", cost_amount)
        return

    # Block level: $100 hit → auto-revoke key + disable Cloud Run
    if cost_amount >= BLOCK_BUDGET:
        print("🚫 BLOCK threshold hit — executing kill switch")
        _revoke_api_key()
        if CLOUD_RUN_SVC:
            _disable_cloud_run_service()
        _create_monitoring_incident("CRITICAL", cost_amount)

def _revoke_api_key():
    try:
        client = apikeys_v2.ApiKeysClient()
        name = f"projects/{PROJECT_ID}/locations/global/keys/{KEY_ID}"
        key = client.get_key(name=name)
        # Remove all API restrictions = effectively disables the key
        key.restrictions.ClearField("api_targets")
        client.update_key(key=key, update_mask={"paths": ["restrictions"]})
        print(f"✅ API Key {KEY_ID} successfully revoked")
    except Exception as e:
        print(f"❌ Key revocation failed: {e}")
        raise

GitHub Leak Scanner — Cloud Scheduler job that searches GitHub for your exposed API keys and auto-revokes them. 
# === GitHub API Key Leak Scanner ===
# Cloud Scheduler → Cloud Run job (runs every 5 minutes)

import os, re, requests, logging
from google.cloud import apikeys_v2, secretmanager

logging.basicConfig(level=logging.INFO)

PROJECT_ID    = os.environ["PROJECT_ID"]
GITHUB_TOKEN  = os.environ["GITHUB_TOKEN"]
MANAGED_KEYS  = os.environ.get("MANAGED_KEY_IDS", "").split(",")

# GCP API key pattern (AIza... format)
GCP_KEY_PATTERN = re.compile(r'AIza[0-9A-Za-z\-_]{35}')

def get_current_key_values() -> dict:
    """Fetch actual key strings from Secret Manager"""
    client = secretmanager.SecretManagerServiceClient()
    key_map = {}
    for key_id in MANAGED_KEYS:
        if not key_id: continue
        secret_name = f"projects/{PROJECT_ID}/secrets/apikey-{key_id}/versions/latest"
        resp = client.access_secret_version(name=secret_name)
        key_val = resp.payload.data.decode()
        key_map[key_val] = key_id
    return key_map

def search_github_for_key(key_prefix: str) -> list:
    """Search GitHub code for the first 16 chars of the key"""
    query = f"{key_prefix[:16]} language:python OR language:javascript OR language:yaml"
    headers = {
        "Authorization": f"token {GITHUB_TOKEN}",
        "Accept": "application/vnd.github.v3+json"
    }
    resp = requests.get(
        "https://api.github.com/search/code",
        params={"q": query, "per_page": 5},
        headers=headers,
        timeout=10
    )
    if resp.status_code != 200:
        return []
    return resp.json().get("items", [])

def auto_revoke_leaked_key(key_id: str, leak_url: str):
    """Immediately revoke the key and log the incident"""
    logging.critical(f"🚨 LEAK DETECTED: Key {key_id} found at {leak_url}")
    try:
        client = apikeys_v2.ApiKeysClient()
        key_name = f"projects/{PROJECT_ID}/locations/global/keys/{key_id}"
        request = apikeys_v2.DeleteKeyRequest(name=key_name)
        client.delete_key(request=request)
        logging.critical(f"✅ Key {key_id} DELETED — was leaked on GitHub")
    except Exception as e:
        logging.error(f"❌ Revocation error: {e}")

def main():
    logging.info("🔍 Starting GitHub leak scan...")
    key_map = get_current_key_values()

    for key_val, key_id in key_map.items():
        results = search_github_for_key(key_val)
        if results:
            for item in results:
                leak_url = item.get("html_url", "unknown")
                auto_revoke_leaked_key(key_id, leak_url)
        else:
            logging.info(f"✅ Key {key_id}: No leaks found")

    logging.info("✅ Scan complete")

if __name__ == "__main__":
    main()
